CREATE TABLE customer_org (
    id SERIAL PRIMARY KEY,
    organization_name VARCHAR(255) NOT NULL,
    company_overview TEXT,
    business_activities TEXT,
    history TEXT,
    group_companies TEXT,
    major_business_partners TEXT,
    sales_trends TEXT,
    president_message TEXT,
    interview_articles TEXT,
    past_transactions TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Create people table with foreign key to organization
CREATE TABLE customer_people (
    id SERIAL PRIMARY KEY,
    person_name VARCHAR(255) NOT NULL,
    organization_id INTEGER REFERENCES customer_org(id) ON DELETE CASCADE,
    organization VARCHAR(255),
    title VARCHAR(255),
    career_history TEXT,
    current_activities TEXT,
    publications TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
